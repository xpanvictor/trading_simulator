#include "tradeapp/model/order.hpp"
#include <limits>
#include <stdexcept>
using namespace std;

// max order quantity is max long for now
constexpr long MAX_ORDER_QUANTITY = std::numeric_limits<long>::max();

Order::Order(
    int id, string traderId, long price, long quantity, EOrderType ot)
    : id{id}, traderId{traderId}, price{price}, quantity{quantity}, orderType{ot}
{
    if (quantity < 0 || quantity > MAX_ORDER_QUANTITY)
    {
        throw std::length_error("invalid quantity: " + std::to_string(quantity));
    }
    if (price < 0)
    {
        throw std::invalid_argument("invalid price");
    }
}

long Order::getPrice() { return price; }
long Order::getQuantity() { return quantity; }
